el = (parent, tag) ->
  element = document.createElement tag
  parent.appendChild element
  return element

window.print = (message) ->
  $output = el document.body, 'div'
  $output.style.margin = '0.5em'
  $output.style.padding = '0.5em' 
  $output.style.fontFamily = 'Helvetica'
  $output.style.fontSize = '24px'
  $output.style.backgroundColor = '#000'
  $output.style.color = '#0F0'
  $output.style.borderRadius = '0.5em' 
  $output.innerHTML = message

increase = (x) -> 
  return x + 1

decrease = (x) -> 
  return x - 1

plus = (x, y) -> 
  if x == 0 
    result = y
  else
    result = plus (decrease x),  (increase y)
  print "plus #{x} + #{y} = #{result}"
  return result

window.plusTo = (n) ->
  if n == 1
    result = 1
  else
    result =  plus n, plusTo(decrease n)
  return result
