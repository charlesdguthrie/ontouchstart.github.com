el = (parent, tag) ->
  element = document.createElement tag
  parent.appendChild element
  return element

print = (message) ->
  $output = el document.body, 'div'
  $output.innerHTML = message

print "Hello, world."

print "Time is #{new Date()}."

counter = 10
countDown = () ->
  () =>
    print "[#{new Date()}] #{counter--}"

print "Count down now"
for i in [0..10]
  setTimeout countDown(), i * 1000
