el = (parent, tag) ->
  element = document.createElement tag
  parent.appendChild element
  return element

format = (dt) ->
  hh = Math.floor(dt / 3600)
  mm = Math.floor(dt / 60) - hh * 60
  ss = Math.floor(dt) - mm * 60 - hh * 3600
  hh = "0#{hh}" if hh < 10
  mm = "0#{mm}" if mm < 10
  ss = "0#{ss}" if ss < 10
  return "#{dt.toFixed(2)} [#{hh}:#{mm}:#{ss}]"

window.timer = (limit) =>
  $output = el document.body, 'div'

  $output.style.width = '12em'
  $output.style.margin = 'auto'
  $output.style.marginBottom = '0.5em'
  $output.style.padding = '0.5em' 
  $output.style.fontFamily = 'Helvetica'
  $output.style.fontSize = '24px'
  $output.style.textAlign = 'right'
  $output.style.backgroundColor = '#000'
  $output.style.color = '#0F0'
  $output.style.borderRadius = '0.5em'

  update = (message) => $output.innerHTML = message

  d0 = new Date
  dt = 0

  refresh = () =>
    dt = (limit * 1000 - (new Date() - d0))/1000
    if dt < 0
      clearInterval(timer)
      dt = 0
    update format dt

  timer = setInterval refresh, 1
