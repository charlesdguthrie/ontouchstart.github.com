el = (parent, tag) ->
  element = document.createElement tag
  parent.appendChild element
  return element

window.print = (message) ->
  $output = el document.body, 'div'
  $output.className = 'output'
  $output.innerHTML = message

increase = (x) -> 
  return x + 1

decrease = (x) -> 
  return x - 1

window.plus = (x, y) -> 
  if x == 0 
    result = y
  else
    result = plus (decrease x),  (increase y)
  print "plus #{x} + #{y} = #{result}"
  return result
