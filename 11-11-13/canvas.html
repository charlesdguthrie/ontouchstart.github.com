cs '''
setCanvas= () ->
  c = el document.body, 'canvas'
  c.width = 1024
  c.height = 1024
  c.style.backgroundColor = '#000'
  ctx = c.getContext("2d")

  document.ontouchstart = (e) ->
    x = e.touches[0].pageX
    y = e.touches[0].pageY
    r = Math.floor(Math.random() * 255)
    g = Math.floor(Math.random() * 255)
    b = Math.floor(Math.random() * 255)
    ctx.lineWidth = 2
    ctx.restore()
    ctx.strokeStyle = 'rgba(' + r + ',' + g + ',' + b + ',1)'
    ctx.beginPath()
    ctx.moveTo x, y

  document.ontouchmove = (e) ->
    e.preventDefault();
    x = e.touches[0].pageX
    y = e.touches[0].pageY
    ctx.lineTo x, y
    ctx.stroke()

  document.ontouchend = (e) ->
    ctx.save()

setCanvas()
'''
