el = (parent, tag) ->
  element = document.createElement tag
  parent.appendChild element
  return element

window.print = (message) ->
  $output = el document.body, 'div'
  $output.className = 'output'
  $output.innerHTML = message

remainder = (a, b) ->
  return a if a < b
  remainder a-b, b
window.gcd = (a, b) ->
  return a if b == 0
  gcd b, remainder a, b

for i in [1..10]
  for j in [1..10]
    result = gcd i, j
    print "gcd #{i}, #{j} = #{result}"
