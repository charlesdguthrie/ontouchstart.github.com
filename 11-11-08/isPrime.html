el = (parent, tag) ->
  element = document.createElement tag
  parent.appendChild element
  return element

window.print = (message) ->
  $output = el document.body, 'div'
  $output.className = 'output'
  $output.innerHTML = message

remainder = (a, b) ->
  return a if a < b
  remainder a-b, b
gcd = (a, b) ->
  return a if b == 0
  gcd b, remainder a, b
window.isPrime = (n) ->
  return true if n == 2
  for i in [2..n-1]
    x = gcd n, i
    print "gcd #{n}, #{i} = #{x}"
    return false if  x > 1
  return true
