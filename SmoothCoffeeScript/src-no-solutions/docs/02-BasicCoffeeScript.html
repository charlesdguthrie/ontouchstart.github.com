Conditionals
             		                            
  show '--- Conditionals ---'
  for counter in [0..20]
    if counter % 3 == 0 and counter % 4 == 0
      show counter

  for counter in [0..20]
    if counter % 4 == 0
      show counter
    if counter % 4 != 0
      show '(' + counter + ')'

  for counter in [0..20]
    if counter % 4 == 0
      show counter
    else
      show '(' + counter + ')'

  for counter in [0..20]
    if counter > 15
      show counter + '**'
    else if counter > 10
      show counter + '*'
    else
      show counter
