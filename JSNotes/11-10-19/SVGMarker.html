function el(parent, tag, html) {
  var element = document.createElement(tag);
  if(html) element.innerHTML = html;
  parent.appendChild(element);
  return element;
}

function marker() {
  var bgSrc = 'Marker.svg';
  var markerDiv = el(document.body, 'div');
  markerDiv.style.width = '612px';
  markerDiv.style.height = '396px';
  markerDiv.style.margin = 'auto';
  markerDiv.style.backgroundSize = '612px 396px';
  markerDiv.style.background = '#abc url(' + bgSrc + ')';
  markerDiv.onclick = function () {
    document.location = 'http://rletkeman.blogspot.com/2011/08/marker-tutorial.html';
  };
}

marker();
